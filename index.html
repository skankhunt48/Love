/**
* 曾经你期待过一个承诺,
* 希望这个礼物能弥补过去的遗憾.
*/
Girl u = new Girl("Jingwei");
// Jan 25, 1999, 一个新的生命呱呱坠地.
u.liveHappily();
// 你拥有一个幸福的家庭，快乐的童年，优秀的学习成绩.
u.cameToBeijing();
// Sep 1,2017，你进入北京师范大学，开始了新的生活.
Boy i = new Boy("Xingyu");
i.met(u);
// Jan 7, 2022，我们第一次在金谷园见面.
// 你似乎很辛苦，总有写不完的作业和论文.
boolean isBusy = true;
while (isBusy) {
i.amWith(u);
// 我试图用一些崭新的意向去形容你.
// 比如那些在我认知之外的事物.
// 比如我明天的经历，猎户座星云的位置，以及天山山麓冰川移动的方向.
// 正如时，间，节这些错综复杂，循环往复的概念.
// 永恒绵延的流逝感，是我们内心最害怕的东西.
// 于是记忆里的一切，都被赋予了同样的光.
isBusy = u.graduate();
}
// 我不知道今天的月亮是否和昨天不一样.
// 可我祝愿你能够寻找到某种人生的意义.
i.wish(u);
u.liveHappilyForever();
